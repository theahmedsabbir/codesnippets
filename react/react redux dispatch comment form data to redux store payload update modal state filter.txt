export const Reducer = (state = initialState, action) => {
    // console.log('from reducer', action);
    if (action.type === 'ADD_COMMENT') {


        // first comment ta nilam
        let comment = action.payload;

        // comment e dish id add
        let dish_id = action.dish_id;

        // jei dish ta replace korte hobe nie ashlam 
        const dish = state.dishes.filter(dish =>
            dish.id === dish_id)[0];

        //notun comment er id dilam
        comment.id = dish.comments.length;

        // dish er comment gula update korlam notun comment die
        dish.comments = [...dish.comments, comment];

        // state update korlam 
        let stateCopy = { ...state };
        stateCopy.dishes[dish.id] = dish;


        // return this state
        return stateCopy;
    }
    return state;
}
